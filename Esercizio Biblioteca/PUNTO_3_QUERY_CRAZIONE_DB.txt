CREATE TABLE utenti (
    id_utente INT PRIMARY KEY AUTO_INCREMENT,
    nome VARCHAR(255) NOT NULL,
    cognome VARCHAR(255) NOT NULL,
    cf VARCHAR(16) NOT NULL UNIQUE,
    email VARCHAR(255) NOT NULL UNIQUE,
    username VARCHAR(255) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL,
    flag INT NOT NULL DEFAULT 1
);

CREATE TABLE libri (
    id_libro INT PRIMARY KEY AUTO_INCREMENT,
    titolo VARCHAR(255) NOT NULL,
    autore VARCHAR(255) NOT NULL,
    editore VARCHAR(255) NOT NULL,
    prezzo DECIMAL(10,2) NOT NULL,
    isbn VARCHAR(13) NOT NULL UNIQUE,
    n_pagine INT NOT NULL,
    genere VARCHAR(255) NOT NULL,
    disponibile VARCHAR(2) NOT NULL DEFAULT 'SI'
);

CREATE TABLE prestiti (
    id_prestito INT PRIMARY KEY AUTO_INCREMENT,
    id_utente INT NOT NULL,
    id_libro INT NOT NULL,
    data_inizio DATE,
    data_fine DATE,
    data_restituzione DATE,
    giorni_richiesti INT,
    stato ENUM('IN_ATTESA', 'IN_PRESTITO', 'TERMINATO', 'RIFIUTATO') NOT NULL,
    motivazione_rifiuto VARCHAR(255),
    FOREIGN KEY (id_utente) REFERENCES utenti(id_utente),
    FOREIGN KEY (id_libro) REFERENCES libri(id_libro)
);
